Python menyediakan library yang kuat untuk melakukan permintaan HTTP. Gunakan library requests untuk implementasi sederhana.

TypeScript dapat digunakan dengan fetch di Node.js atau browser. Gunakan variabel lingkungan untuk mengamankan API key Anda.

with requests.post(url, json=payload, headers=headers, stream=True) as response:
    response.raise_for_status()
    for chunk in response.iter_lines():
        if chunk:
            # Process each chunk (which is a server-sent event)
            print(chunk.decode('utf-8'))

import fetch from 'node-fetch';

const apiKey = process.env.CALISTA;

if (!apiKey) {
  throw new Error('CALISTA environment variable not set');
}

const url = 'https://api.calista.ai/chat/completions';

const options = {
  method: 'POST',
  headers: {
    accept: 'application/json',
    'content-type': 'application/json',
    Authorization: `Bearer ${apiKey}`,
  },
  body: JSON.stringify({
    model: 'llama3.2:3b',
    messages: [
      { role: 'user', content: 'How do I use TypeScript for API calls?' },
    ],
  }),
};

async function makeRequest() {
  try {
    const response = await fetch(url, options);
    const data = await response.json();
    console.log(data);
  } catch (error) {
    console.error('Error:', error);
  }
}

makeRequest();

import fetch from 'node-fetch';

const apiKey = process.env.CALISTA;

if (!apiKey) {
  throw new Error('CALISTA environment variable not set');
}

const url = 'https://api.calista.ai/chat/completions';

const options = {
  method: 'POST',
  headers: {
    accept: 'application/json',
    'content-type': 'application/json',
    Authorization: `Bearer ${apiKey}`,
  },
  body: JSON.stringify({
    model: 'llama3.2:3b',
    stream: true,
    messages: [
      { role: 'user', content: 'How do I stream responses in TypeScript?' },
    ],
  }),
};

async function makeStreamingRequest() {
  try {
    const response = await fetch(url, options);
    if (!response.body) {
        console.log("No response body");
        return;
    }
    for await (const chunk of response.body) {
      console.log(chunk.toString());
    }
  } catch (error) {
    console.error('Error:', error);
  }
}

makeStreamingRequest();
